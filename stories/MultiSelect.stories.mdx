# MultiSelect

MultiSelect component allows user to make a multiple selection from a list of options.

export const Template = (args) => {
  const [value, setValue] = useState();
  const onSelect = (value) => {
    setValue(value);
  };
  return <MultiSelect {...args} value={value} onSelect={onSelect} />;
};
